
```python
# Adição
a = 5.2
b = 3.1
result = a + b
print(result)  # Saída: 8.3

# Subtração
c = 7.5
d = 4.2
result = c - d
print(result)  # Saída: 3.3

# Multiplicação
e = 2.5
f = 6.4
result = e * f
print(result)  # Saída: 16.0

# Divisão
g = 10.0
h = 2.0
result = g / h
print(result)  # Saída: 5.0

# Divisão Inteira (o result sempre será um float)
i = 10.0
j = 3.0
result = i // j
print(result)  # Saída: 3.0

# Módulo
k = 10.5
l = 3.2
result = k % l
print(result)  # Saída: 1.7

# Potenciação
m = 2.0
n = 3.5
result = m ** n
print(result)  # Saída: 11.313708498984761

```

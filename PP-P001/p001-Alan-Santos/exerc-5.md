#Adicao e atribuição

a = 5
b = 3
a += b
print(a)  # Saída: 8

#Subtração e atribuição

a = 5
b = 3
a -= b
print(a)  # Saída: 2

#Multiplicação e atribuição

a = 5
b = 3
a *= b
print(a)  # Saída: 15

#Divisão e atribuição

a = 5
b = 2
a /= b
print(a)  # Saída: 2.5

#Divisão inteira e atribuição

a = 5
b = 2
a //= b
print(a)  # Saída: 2

#Resto da divisão e atribuição

a = 5
b = 2
a %= b
print(a)  # Saída: 1

#Potenciação e atribuição

a = 2
b = 3
a **= b
print(a)  # Saída: 8
